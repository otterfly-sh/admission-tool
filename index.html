const SHEET_ID = '1S5bNJtaEJej2YnFwkYG0X9p2zXh395njUASfQaIn5O4'; const ADMIT_SHEET = '錄取資料'; const STUDENT_SHEET = '學生名單'; const SCHOOL_SHEET = '學校清單'; const ABBR_SHEET = '學校簡稱'; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); let sheet = ss.getSheetByName(ADMIT_SHEET); if (!sheet) { sheet = ss.insertSheet(ADMIT_SHEET); sheet.appendRow(['學生姓名','學校','獎學金','填入時間','填入者']); } sheet.appendRow([ data.studentName, data.schoolName, data.scholarship || '', new Date().toLocaleString('zh-TW'), data.filledBy || '' ]); return buildResponse({status:'ok'}); } catch(err) { return buildResponse({status:'error', message: err.toString()}); } } function doGet(e) { try { const action = e.parameter.action || 'getAdmissions'; const ss = SpreadsheetApp.openById(SHEET_ID); if (action === 'getStudents') { const sheet = ss.getSheetByName(STUDENT_SHEET); if (!sheet) return buildResponse({students:[]}); const rows = sheet.getDataRange().getValues(); const headers = rows[0].map(h => h.toString().trim().toLowerCase()); const nameIdx = headers.findIndex(h => /studentname|姓名/i.test(h)); const gpaIdx = headers.findIndex(h => /gpa/i.test(h)); const satIdx = headers.findIndex(h => /sat/i.test(h)); const engIdx = headers.findIndex(h => /english|toefl|englishproficiency/i.test(h)); const students = rows.slice(1) .filter(r => r[nameIdx] && r[nameIdx].toString().trim() !== '') .map(r => ({ name: r[nameIdx].toString().trim(), gpa: gpaIdx >= 0 ? r[gpaIdx].toString() : '', sat: satIdx >= 0 ? r[satIdx].toString() : '', english: engIdx >= 0 ? r[engIdx].toString() : '' })); return buildResponse({students}); } if (action === 'getSchools') { const sheet = ss.getSheetByName(SCHOOL_SHEET); if (!sheet) return buildResponse({schools:[]}); const rows = sheet.getDataRange().getValues(); const headers = rows[0].map(h => h.toString().trim().toLowerCase()); const rankIdx = headers.findIndex(h => /rank/i.test(h)); const nameIdx = headers.findIndex(h => /name|學校/i.test(h)); const schools = rows.slice(1) .filter(r => r[nameIdx] && r[nameIdx].toString().trim() !== '') .map(r => ({ rank: rankIdx >= 0 ? r[rankIdx].toString() : '', name: r[nameIdx].toString().trim() })); return buildResponse({schools}); } if (action === 'getAbbreviations') { const sheet = ss.getSheetByName(ABBR_SHEET); if (!sheet) return buildResponse({abbreviations:[]}); const rows = sheet.getDataRange().getValues(); const headers = rows[0].map(h => h.toString().trim()); const nameIdx = headers.findIndex(h => /學校名稱|name/i.test(h)); const abbrIdx = headers.findIndex(h => /簡稱|abbr/i.test(h)); const abbreviations = rows.slice(1) .filter(r => r[nameIdx] && r[abbrIdx]) .map(r => ({ name: r[nameIdx].toString().trim(), abbr: r[abbrIdx].toString().trim() })); return buildResponse({abbreviations}); } if (action === 'getAdmissions') { const sheet = ss.getSheetByName(ADMIT_SHEET); if (!sheet) return buildResponse({data:[]}); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const data = rows.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i] ? row[i].toString() : ''); return obj; }); return buildResponse({data}); } return buildResponse({error:'unknown action'}); } catch(err) { return buildResponse({error: err.toString()}); } } function buildResponse(obj) { const output = ContentService.createTextOutput(JSON.stringify(obj)); output.setMimeType(ContentService.MimeType.JSON); return output; }
